# Включение НАСТОЯЩЕЙ авто-отправки отчётов по SMTP.
# Чтобы включить: переименуй этот файл в  RevitQuickAccess_mail.txt  и положи рядом с
# RevitQuickAccess.dll  (папка %AppData%\Autodesk\Revit\Addins\2026).
# Тогда плагин будет сам слать отчёты (в т.ч. авто-отправлять отчёты о крашах на следующем запуске).
# Без этого файла кнопка "человечек" открывает почтовый клиент с готовым письмом (без паролей).
#
# from/password — это аккаунт-ОТПРАВИТЕЛЬ (не обязательно тот же, куда шлём).
# Для mail.ru нужен ПАРОЛЬ ПРИЛОЖЕНИЯ (не основной пароль): создаётся в настройках почты.
# mail.ru обычно: host=smtp.mail.ru, port=587 (STARTTLS). Если 587 не работает — пробуй 2525.

host=smtp.mail.ru
port=587
from=твой_отправитель@mail.ru
